3)SHEMA DE LA BASE DE DONNEE(mise a jour)

Base de donnée : agence

Tables:
A)CREATE TABLE `bien` (
  `idBien` int(11) NOT NULL,
  `Type` enum('Appartement','Villa') DEFAULT NULL,
  `Nom` varchar(50) DEFAULT NULL,
  `Prix` int(10) UNSIGNED DEFAULT NULL,
  `Adresse_Position` varchar(210) DEFAULT NULL,
  `Description` varchar(255) DEFAULT NULL,
  `likes` int(11) DEFAULT 0,
  `dislikes` int(11) DEFAULT 0
) ENGINE=InnoDB DEFAULT CHARSET=armscii8 COLLATE=armscii8_bin;

B)
CREATE TABLE `client` (
  `idCli` int(11) NOT NULL,
  `Adresse` varchar(150) DEFAULT NULL,
  `Nom` varchar(50) DEFAULT NULL,
  `Prenom` varchar(250) DEFAULT NULL,
  `Telephone` varchar(20) DEFAULT NULL,
  `mot_de_passeCl` varchar(200) DEFAULT NULL,
  `genre` enum('Homme','Femme') DEFAULT NULL,
  `Age` int(10) UNSIGNED DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=armscii8 COLLATE=armscii8_bin;

C)
CREATE TABLE `commerciaux` (
  `idCom` int(11) NOT NULL,

  `Telephone` varchar(20) DEFAULT NULL,
  `Profession` varchar(150) DEFAULT NULL,
  `mot_de_passeC` varchar(200) DEFAULT NULL,
  `Genre` enum('Homme','Femme') DEFAULT NULL,
  `Age` int(10) UNSIGNED DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=armscii8 COLLATE=armscii8_bin;
